#include <iostream>
using namespace std;
int linearSearch(int data, int arr[], int sizeOfArray )
{
    for (int i = 0; i < sizeOfArray; i++)
    {
        if (arr[i] == data)
        {
            return i;
        }
    }
    return -1;
}
int main()
{

    int arr[10] = { 34,55,77,98,30,37,12,15,78,3 };
    int sizeOfArray = 10;
    int data;
    cout << "Enter the data which you want to search: ";
    cin >> data;
    int index;
    index=linearSearch(data, arr, sizeOfArray);
    if (index == -1)
    {
        cout << "Ooops! Inputted data has not been found." << endl;
    }
    else
    {
        cout << "Inputted data has been found at " << index << " no index" << endl;
    }
    return 0;

}
